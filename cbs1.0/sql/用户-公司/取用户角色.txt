select bui.org_id,
       bso.name,
       bsr.name,
       (CASE bsr.type
           WHEN '1' THEN
               '公司'
           WHEN '0' THEN
               '通用'
           ELSE
               ''
           END) as 角色类型
        ,
       bui.*
from prd_dl_base.base_user_info bui
         left join prd_dl_base.base_sys_org bso
                   on bso.id = bui.org_id
         inner join prd_dl_base.base_sys_user_role bsur
                    on bsur.user_id = bui.id
         inner join prd_dl_base.base_sys_role bsr
                    on bsr.id = bsur.role_id
         left join prd_dl_base.base_sys_org bso2
                   on bso2.code = bsr.org_code
where bui.account = 'huanglei'
  and bsr.tenant_type = '5'
  and bsr.status = '1'
